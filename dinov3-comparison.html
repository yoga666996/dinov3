ü¶ï DINOv3 Usage Example

# DINOv3 Image Classification
from transformers import Dinov2Model, Dinov2ImageProcessor
import torch

# Load model
processor = Dinov2ImageProcessor.from_pretrained(
    "facebook/dinov2-large")
model = Dinov2Model.from_pretrained(
    "facebook/dinov2-large")

# Inference
inputs = processor(images=image, return_tensors="pt")
with torch.no_grad():
    outputs = model(**inputs)
features = outputs.pooler_output

# Feature dimensions: [1, 1024]
print(f"Features shape: {features.shape}")

üìé CLIP Usage Example

# CLIP Image-Text Matching
import clip
import torch

# Load model
model, preprocess = clip.load("ViT-L/14")

# Prepare inputs
image = preprocess(image).unsqueeze(0)
text = clip.tokenize(["a photo of a cat"])

# Calculate similarity
with torch.no_grad():
    logits_per_image, logits_per_text = model(image, text)
    probs = logits_per_image.softmax(dim=-1)

print(f"Similarity: {probs[0][0]:.3f}")

# Model Inference Speed Comparison
import time
import numpy as np

def benchmark_model(model, processor, images, runs=100):
    times = []
    for _ in range(runs):
        start = time.time()
        inputs = processor(images=images, return_tensors="pt")
        with torch.no_grad():
            _ = model(**inputs)
        times.append(time.time() - start)
    
    return {
        'mean': np.mean(times),
        'std': np.std(times),
        'fps': 1.0 / np.mean(times)
    }
